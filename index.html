PASOS PARA EJECUTAR EL BOT EN TU PC
1

Crear tu bot en Telegram

Buscá @BotFather en Telegram → escribí /newbot → elegí nombre → copiá el token que te da.

2

Obtener tu Chat ID

Buscá @userinfobot en Telegram → mandá cualquier mensaje → copiá tu Chat ID.

3

Instalar Python

Descargá desde python.org. En Windows marcá "Add to PATH" durante la instalación.

4

Instalar librerías

Abrí la terminal (cmd en Windows) y ejecutá:

pip install yfinance pandas requests schedule
5

Copiar el código del bot

Copiá el código de abajo → pegalo en un archivo balanzbot.py → completá tu token y chat ID donde dice TU_TOKEN_AQUI.

6

Ejecutar el bot

En la terminal, navegá a la carpeta y ejecutá python balanzbot.py. El bot queda corriendo y te llegan alertas automáticamente.

CÓDIGO PYTHON — COPIAR Y PEGAR
# BalanzBot — Bot de alertas de trading para Balanz # Instalá: pip install yfinance pandas requests schedule import yfinance as yf import requests, schedule, time from datetime import datetime TOKEN = "TU_TOKEN_AQUI" # Del BotFather CHAT_ID = "TU_CHAT_ID_AQUI" # De @userinfobot WATCHLIST = [ "GGAL.BA", "YPFD.BA", "BBAR.BA", # Acciones arg "AAPL", "GOOGL", "MSFT", # CEDEARs "BTC-USD", "ETH-USD", # Crypto ] RSI_COMPRA = 30 # señal de compra RSI_VENTA = 70 # señal de venta RSI_WATCH = 38 # alerta previa a compra MMA_C = 9 # media móvil corta MMA_L = 21 # media móvil larga INTERVALO = 15 # minutos entre escaneos def alerta(msg): requests.post( f"https://api.telegram.org/bot{TOKEN}/sendMessage", json={"chat_id": CHAT_ID, "text": msg, "parse_mode": "HTML"} ) def calcular_rsi(serie, periodo=14): d = serie.diff() g = d.clip(lower=0).rolling(periodo).mean() l = (-d.clip(upper=0)).rolling(periodo).mean() return 100 - 100 / (1 + g / l) def analizar(ticker): try: df = yf.download(ticker, period="3mo", interval="1d", progress=False) if df.empty or len(df) < 30: return c = df["Close"].squeeze() rsi = calcular_rsi(c).iloc[-1] mc = c.rolling(MMA_C).mean().iloc[-1] ml = c.rolling(MMA_L).mean().iloc[-1] px = c.iloc[-1] h = datetime.now().strftime("%H:%M") if rsi < RSI_COMPRA and mc > ml: alerta(f"🟢 <b>COMPRAR {ticker}</b>\n" f"RSI={rsi:.1f} | Precio=${px:.2f}\n" f"✅ Cruce alcista confirmado | {h}") elif rsi > RSI_VENTA and mc < ml: alerta(f"🔴 <b>VENDER {ticker}</b>\n" f"RSI={rsi:.1f} | Precio=${px:.2f}\n" f"⚠️ Cruce bajista confirmado | {h}") elif rsi < RSI_WATCH: alerta(f"🟡 <b>ATENCIÓN {ticker}</b>\n" f"RSI={rsi:.1f} — acercándose a zona de compra\n" f"Precio=${px:.2f} | {h}") except Exception as e: print(f"Error {ticker}: {e}") def escanear(): print(f"\n[{datetime.now():%H:%M}] Escaneando {len(WATCHLIST)} activos...") for t in WATCHLIST: analizar(t) # Inicio alerta("🤖 <b>BalanzBot activo</b> — Monitoreando el mercado.") escanear() schedule.every(INTERVALO).minutes.do(escanear) while True: schedule.run_pending() time.sleep(60)
